# Step 5: Update CMakeLists.txt
In viospice/cli/CMakeLists.txt, add:

```cmake
# FluxScript integration
set(VIO_CMD_SOURCES
    ${VIO_CMD_SOURCES}
    flux_command.cpp
)

# Link with FluxScript if available
if(FLUXSCRIPT_FOUND)
    target_link_libraries(vio-cmd PRIVATE FluxScript)
    target_compile_definitions(vio-cmd PRIVATE HAVE_FLUXSCRIPT)
endif()
```

# Step 6: Add CMake option for FluxScript
In viospice/CMakeLists.txt, add:

```cmake
option(VIOSPICE_ENABLE_FLUXSCRIPT "Enable FluxScript integration" ON)

if(VIOSPICE_ENABLE_FLUXSCRIPT)
    find_package(FluxScript QUIET)
    if(FluxScript_FOUND)
        set(FLUXSCRIPT_FOUND TRUE)
        include_directories(${FLUXSCRIPT_INCLUDE_DIRS})
    endif()
endif()
```
